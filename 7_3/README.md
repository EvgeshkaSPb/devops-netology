В уже созданный aws_instance добавьте зависимость типа инстанса от вокспейса, что бы в разных ворскспейсах использовались разные instance_type.
Добавим count. Для stage должен создаться один экземпляр ec2, а для prod два.
Создайте рядом еще один aws_instance, но теперь определите их количество при помощи for_each, а не count.
Что бы при изменении типа инстанса не возникло ситуации, когда не будет ни одного инстанса добавьте параметр жизненного цикла create_before_destroy = true в один из рессурсов aws_instance.

student@student-virtual-machine:~/aws-terraform$ nano main.tf

provider "aws" {
    region = "eu-central-1"
}

terraform {
    backend "s3" {
        bucket = "study-bucket-1-homework"
        key = "global/s3/terraform.tfstate"
        region = "eu-central-1"
        dynamodb_table = "study-terraform-state-locks"

data "aws_ami" "ubuntu" {
  most_recent = true

  filter {
    name   = "name"
    values = ["ubuntu/images/hvm-ssd/ubuntu-focal-20.04-amd64-server-*"]
  }

  filter {
    name   = "virtualization-type"
    values = ["hvm"]
  }

  owners = ["099720109477"]
}

locals {
  type_ws = {
    stage = "t3.micro"
    prod = "t3.large"
  }
  count_vm = {
    stage = 1
    prod = 2
  }
}

resource "aws_instance" "count" {
  ami           = data.aws_ami.ubuntu.id
  instance_type =  local.type_ws[terraform.workspace]
  count = local.count_vm[terraform.workspace]
  tags = {
  name = "HW-${count.index+1}"
  }
}

locals {
  vm = {
    prod = {
      vm1 = "t3.large"
      vm2 = "t3.large"
      }
    stage = {
      vm3 = "t3.micro"
      }
  }
}

resource "aws_instance" "for_each" {
  for_each = local.vm [terraform.workspace]
  ami           = data.aws_ami.ubuntu.id
  instance_type =  each.value
  tags = {
  name = "HW-${each.key}"
  }
  lifecycle {
    create_before_destroy = true
  }
}
